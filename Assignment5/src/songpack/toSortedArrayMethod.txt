            private ArrayList<Song> toSortedArrayList(ArrayList<Song> list, Node root){

                if(root == null)
                    return list;

                list.addAll(toSortedArrayList(list, root.left));
                list.add(root.data);
                list.addAll(toSortedArrayList(list, root.right));

                return list;
            }
